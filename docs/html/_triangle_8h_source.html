28 constexpr float epsilon = Math::Constants::Epsilon<float>;
29
30 Math::Vector3f rayCrossEdge2 = Math::Cross(ray.direction, edges[1]);
31 float determinant = Math::Dot(edges[0], rayCrossEdge2);
32
33 if (Math::Abs(determinant) < epsilon) {
34 return false;
35 }
36
37 float inverseDeterminant = 1.f / determinant;
38 Math::Vector3f s = ray.origin - vertices[0];
39 float u = inverseDeterminant * Math::Dot(s, rayCrossEdge2);
40
41 if (u < 0.f || u > 1.f) {
42 return false;
43 }
44
45 Math::Vector3f sCrossEdge1 = Math::Cross(s, edges[0]);
46 float v = inverseDeterminant * Math::Dot(ray.direction, sCrossEdge1);
47
48 if (v < 0.f || u + v > 1.f) {
49 return false;
50 }
51
52 float t = inverseDeterminant * Math::Dot(edges[1], sCrossEdge1);
53
54 if (t < tMin || tMax < t) {
55 return false;
56 }
57
58 payload.t = t;
59 payload.normal = normal;
60 payload.material = material;
